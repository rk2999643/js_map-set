Q.2 "Problem: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps)

                let w= "abcadeecfb";
let map = new Map();
for(let i=0;w.length>i;i++){
    if(map.has(w[i]))
     map.set(w[i],Number(map.get(w[i]))+1);
    else
      map.set(w[i],1);
}
console.log(...map);
